#include<bits/stdc++.h>   //差分数组法（区间表示）
using namespace std;
int main(){
	int l,m,sta,end;
	int tree[1006]={0};
	cin>>l>>m;
	for(int i=0;i<m;i++){
		cin>>sta>>end;
		tree[sta]=1;tree[end]=-1;
		
	}
	int sum=0,b=0;
	for(int i=0;i<=l;i++){
		if(tree[i]==1) b++;
		else if(tree[i]==-1) b--;
		
		if(tree[i]==0 && b==0) sum++;
		
		
		
	}
	
	cout<<sum;
	return 0;
	
}
